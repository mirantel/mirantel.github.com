Функции

@function flex-grid($columns) {
    $width: $columns * $column + ($columns - 1) * $gutter;
    @return percentage($width / $grid-full-width);
}

Миксины

@mixin column($column-num) {
    float: left;
    margin-left: $grid-gutter;
    width: flex-grid($column-num);
}

}

Extend

%field-error {
    border-color: $INPUT_ERROR_COLOR;

    &:focus {
        box-shadow: 0 0 5px 2px lighten($INPUT_ERROR_COLOR, 20%);
    }
}

.input,
.textarea,
.select {
    &.ng-invalid.ng-dirty {
        @extend %field-error;
    }
}

  account: E001,
  add: E002,
  address-book: E003,

@function variable-icon-font($icon-code){
    @return #{"\"\\"}#{$icon-code + "\""};
}

@each $name, $char in $FONT_ICONS {
    .#{$ICONS_CLASS}-#{$name}:before {
        content: variable-icon-font($char);

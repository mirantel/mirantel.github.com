Gruntfile.js


module.exports = function(grunt) {
  'use strict';

  grunt.initConfig({

    stylus: {
      options: {
        compress: false,
        firebug: false,
      },
      all: {
        options: {
          use: [
            function() {
              return require('autoprefixer-stylus')('last 3 version', 'ie >= 8');
            }
          ],
        },
        files: {
          'css/global.css': 'styl/global.styl'
        },
      },
    },
    csso: {
      files: {
        src: 'css/global.css',
        dest:'css/global.min.css'
      }
    },
    watch: {
      styl: {
        files: ['styl/**/*.styl'],
        tasks: ['stylus:all', 'csso']
      },
      livereload: {
        options: { livereload: true },
        files: ['*.html','css/**/*.css','js/**/*.js'],
      },
    },
  });

  grunt.loadNpmTasks('autoprefixer-stylus');
  grunt.loadNpmTasks('grunt-contrib-stylus');
  grunt.loadNpmTasks('grunt-contrib-watch');
  grunt.loadNpmTasks('grunt-csso');

  grunt.registerTask('default', ['stylus','csso','watch']);
};

Важные команды

Устанавливает все зависимости
(в нашем случае плагины) из package.json


npm install

Установливает и добавляет в package.json зависимости


npm install {{PACKAGE_NAME}} --save-dev

Разные варианты запуска Grunt


grunt
grunt {{TASK}}
grunt {{TASK}}:{{TARGET}}

Node Packaged Modules

Total Packages: 84 922

Packages. Какие задачи они решают:

  • Запуск препроцессоров: sass, stylus, less, coffeescript.
  • Сжатие css и js: yui, csso, uglify.js.
  • Проверка синтаксиса: csslint, jshint.
  • Запуск юнит-тестов: jasmine, qunit.
  • Обработка картинок и сборка спрайтов: imgo.
  • Include для html: include-replace
  • Автозапуск тасков при изменения в файлах, собствeенный сервер.
  • И многое другое...

Можно писать свои пакеты, включая в них уже готовые.

Autoprefixer

Андрей Ситник, Злые марсиане

Принципы Автопрефиксера

  • Без интерфейса
  • Работа с чистым CSS
  • Самые актуальные префиксы
  • Автообновление

Configuring autoprefixer


grunt.initConfig({
  autoprefixer: {
    options: {
      browsers: ['last 2 version', 'ie 8', 'ie 9'],
      browsers: ['last 1 version', '> 1%', 'Explorer 7'],
      browsers: ['last 1 version', 'BlackBerry 10", 'Android 4'],
    },
    your_target: {

    },
  },
})

grunt.initConfig({
  stylus: {
    options: {
